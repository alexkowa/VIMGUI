\HeaderA{chorizonDL}{C-horizon of the Kola data with missing values}{chorizonDL}
\keyword{datasets}{chorizonDL}
%
\begin{Description}\relax
This data set is the same as the \code{\LinkA{chorizon}{chorizon}} 
data set in package \code{mvoutlier}, except that values below the 
detection limit are coded as \code{NA}.
\end{Description}
%
\begin{Usage}
\begin{verbatim}
data(chorizonDL)
\end{verbatim}
\end{Usage}
%
